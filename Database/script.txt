create table cuenta
(
id_cuenta bigserial primary key,
numero bigint,
nombre varchar(20),
tipo varchar(20),
tipo_detalle varchar(50),
descripcion varchar(50),
saldo decimal(15,5),
divisa varchar(10),
fecha_registro date 
);


create table asiento
(
numero bigint primary key,
fecha_creacion date,
divisa varchar(20)
);

(

debito decimal(15,5),
credito decimal(15,5),
decripcion varchar(50),
impuesto varchar(50),
fecha_factura date,
proveedor varchar(20)
);
